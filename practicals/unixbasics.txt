pwd

echo "Listing files:"
ls -la

echo "Create directory:"
mkdir demo

echo "Create file:"
touch demo/file.txt

echo "Write text:"
echo "Hello Unix" > demo/file.txt

echo "Display file:"
cat demo/file.txt

echo "Search for word:"
grep "Hello" demo/file.txt

echo "Find file:"
find . -name file.txt

echo "Permissions:"
chmod 755 demo/file.txt

echo "Pipe example:"
cat demo/file.txt | wc -w

echo "Background job:"
sleep 5 &
